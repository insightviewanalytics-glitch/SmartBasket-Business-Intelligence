M Query: DimDate
let
  StartDate = #date(2023, 1, 1),
  EndDate = Date.From(DateTime.LocalNow()),
  DayCount = Duration.Days(EndDate - StartDate) + 1,
  Dates = List.Dates(StartDate, DayCount, #duration(1,0,0,0)),
  DateTable = Table.FromList(Dates, Splitter.SplitByNothing()),
  DateCol = Table.RenameColumns(DateTable, {{"Column1", "Date"}}),
  Year = Table.AddColumn(DateCol, "Year", each Date.Year([Date])),
  Month = Table.AddColumn(Year, "MonthNum", each Date.Month([Date])),
  Quarter = Table.AddColumn(Month, "Quarter", each "Q" & Text.From(Date.QuarterOfYear([Date])))
in Quarter
M Query: DimProduct
let
  Source = FactSales,
  Cols = Table.SelectColumns(Source, {"product_name","aisle"}),
  Distinct = Table.Distinct(Cols),
  AddKey = Table.AddIndexColumn(Distinct, "product_key", 1, 1),
  Renamed = Table.RenameColumns(AddKey, {{"aisle","category"}}),
  SubCat = Table.AddColumn(Renamed, "sub_category",
    each if [category] = "Produce" then "Fresh"
    else if [category] = "Dairy" then "Chilled"
    else "Ambient")
in SubCat
M Query: FactSales
let
  Source = Csv.Document(File.Contents("grocery_chain_data.csv")),
  Promoted = Table.PromoteHeaders(Source),
  TypedCols = Table.TransformColumnTypes(Promoted, {
    {"transaction_date", type date},
    {"quantity", Int64.Type},
    {"unit_price", Currency.Type},
    {"final_amount", Currency.Type}}),
  AddSK = Table.AddIndexColumn(TypedCols, "transaction_key", 1, 1),
  FilterNulls = Table.SelectRows(AddSK, each [store_name] <> null)
in FilterNulls
Core Revenue Measures
Base aggregation measures. All other measures build on these. Mark as hidden in the model; surface only the formatted variants.
-- [1] Total Revenue
Total Revenue =
  SUM( FactSales[final_amount] )

-- [2] Revenue YTD
Revenue YTD =
  TOTALYTD( [Total Revenue], DimDate[Date] )

-- [3] Revenue LY (Last Year same period)
Revenue LY =
  CALCULATE(
    [Total Revenue],
    SAMEPERIODLASTYEAR( DimDate[Date] )
  )

-- [4] Revenue YoY Growth %
Revenue YoY % =
  VAR CY = [Total Revenue]
  VAR PY = [Revenue LY]
  RETURN
    IF( ISBLANK(PY) || PY = 0, BLANK(), (CY - PY) / PY )
Customer & Basket Measures
-- [5] Unique Customers
Unique Customers =
  DISTINCTCOUNT( FactSales[customer_id] )

-- [6] Average Basket Value
Avg Basket Value =
  DIVIDE(
    [Total Revenue],
    COUNTROWS( FactSales )
  )

-- [7] Customer Lifetime Value (CLV)
CLV =
  AVERAGEX(
    VALUES( FactSales[customer_id] ),
    CALCULATE( SUM( FactSales[final_amount] ) )
  )

-- [8] Repeat Customer Rate
Repeat Customer Rate =
  VAR Repeat = COUNTX(
    FILTER(
      SUMMARIZE( FactSales, FactSales[customer_id], "@Visits", COUNTROWS(FactSales) ),
      [@Visits] > 1
    ),
    FactSales[customer_id]
  )
  RETURN DIVIDE( Repeat, [Unique Customers] )
Margin & Discount Measures
-- [9] Total Discount Amount
Total Discount =
  SUM( FactSales[discount_amount] )

-- [10] Discount Rate %
Discount Rate % =
  DIVIDE(
    [Total Discount],
    SUM( FactSales[total_amount] )
  )

-- [11] Revenue at Full Price (pre-discount)
Revenue Full Price =
  SUM( FactSales[total_amount] )

-- [12] Discount Erosion Index (vs chain avg)
Discount Erosion Index =
  VAR LocalRate = [Discount Rate %]
  VAR ChainRate = CALCULATE( [Discount Rate %], ALL( DimStore ) )
  RETURN DIVIDE( LocalRate, ChainRate )
Loyalty & Ranking Measures
-- [13] Total Loyalty Points
Total Loyalty Points =
  SUM( FactSales[loyalty_points] )

-- [14] Store Rank by Revenue
Store Revenue Rank =
  RANKX(
    ALL( DimStore[store_name] ),
    [Total Revenue],
    , DESC, Dense
  )

-- [15] Top N Products (dynamic parameter)
Top N Revenue =
  CALCULATE(
    [Total Revenue],
    TOPN(
      [TopN Parameter],
      ALL( DimProduct[product_name] ),
      [Total Revenue], DESC
    )
  )

-- [16] Rolling 3-Month Average Revenue
Revenue 3M Rolling Avg =
  AVERAGEX(
    DATESINPERIOD( DimDate[Date], LASTDATE( DimDate[Date] ), -3, MONTH ),
    [Total Revenue]
  )
